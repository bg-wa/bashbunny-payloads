clear_active_input()
{
    QUACK DELAY 50
    QUACK BACKSPACE
    QUACK DELAY 100
}

enter_and_wait()
{
    if [ -z "$1" ]; then
        WAIT_DELAY=1000
    else
        WAIT_DELAY=$1
    fi
    QUACK DELAY 200
    QUACK ENTER
    QUACK DELAY ${WAIT_DELAY}
}

clear_history_from_terminal()
{
    if [ "$LINES" -gt "0" ]; then
        QUACK STRING head -n -$LINES "~/.bash_history"
        QUACK ENTER
        QUACK DELAY 100
        QUACK STRING history -c
        QUACK ENTER
        QUACK DELAY 100
    else
        QUACK STRING rm "~/.bash_history"
        QUACK ENTER
        QUACK DELAY 1000
        QUACK STRING history -c
        QUACK ENTER
    fi
    echo "HISTORY CLEARED FROM TERMINAL" >> "${DEBUG_FILE}"
}
